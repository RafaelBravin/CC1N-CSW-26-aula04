         function resolverEquacaoSegundoGrau(a, b, c) {
    if (a === 0) {
        console.log("Isso não é uma equação de segundo grau.");
        return;
    }

    const delta = b * b - 4 * a * c;

    console.log("Delta:", delta);

    if (delta < 0) {
        console.log("A equação não possui raízes reais.");
    } else if (delta === 0) {
        const x = -b / (2 * a);
        console.log("A equação possui uma raiz real:");
        console.log("x =", x);
    } else {
        const x1 = (-b + Math.sqrt(delta)) / (2 * a);
        const x2 = (-b - Math.sqrt(delta)) / (2 * a);

        console.log("A equação possui duas raízes reais:");
        console.log("x1 =", x1);
        console.log("x2 =", x2);
    }
}

// Exemplo de uso
resolverEquacaoSegundoGrau(1, -3, 2);
